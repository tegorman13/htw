# Define column defs function 
col_defs <- function(data) {
  
  cols <- colnames(data)
  
  defs <- lapply(cols, function(x) {
    
    if(is.factor(data[[x]])) {
      colDef(sortable = TRUE, filterable = TRUE,minWidth=108) 
    } else {
      colDef(sortable = TRUE, filterable = FALSE,minWidth=90)
    }
    
  })
  
  setNames(defs, cols)
}

  # Pass original data too
