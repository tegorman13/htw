plot_subject_fits <- function(model, subject_code) {
  pattern <- glue("^r_id\\[{subject_code},.*\\]")
  plot <- mcmc_areas(model, prob = .5, regex_pars = c(pattern)) +
            ggtitle(glue("fit for subject #{subject_code}:"))
  return(plot)
}
